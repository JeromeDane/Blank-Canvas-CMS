namespace BlankCanvas;


class Controller {
	
	public $view;
	protected $_context;
	protected $_name;
	
	public function __construct($context = 'system') {
		
		$this->_context = $context;
		
		// determine simple controller name
		$traceStack = debug_backtrace();
		$className = get_class($traceStack[0]['object']);
		$this->_name = strtolower(preg_replace('/Controller$/', '', $className));
		$this->_name = preg_replace('/^.+\\\\/', '', $this->_name);
		
		// include model
		$modelFilePath = BC_ROOT . ($this->_context == 'system' ? '/system/' : '/plugins/' . $this->_context);
		$modelFilePath .= '/models/' . ucfirst($this->_name) . '.model.inc';
		if (is_file($modelFilePath)) require_once $modelFilePath;
		
		// create view
		$this->view = new View($this->_name);
		
	}
	
}

namespace BlankCanvas;

/**
 * Content
 *
 * System content class
 *
 * @author Jerome
 *
 * @property 	string 	$name	The name of the content item
 * @property	string	$email	The user's email (and login)
 *
 */
class Content extends \BlankCanvas\Model {
	
	public function getFields() {
		return array(
			'parent_id' => array(
				'type'     => 'int',
				'index'    => true,
				'unsigned' => true
			),
			'type'      => array(
				'type'  => 'string',
				'index' => true
			),
			'order'      => array(
				'type'     => 'int',
				'unsigned' => true
			),
			'name'      => array(
				'type'     => 'string',
				'required' => true
			),
			'body'      => array(
				'type' => 'text',
			),
			'created'      => array(
				'type' => 'datetime',
			),
		);
	}
}
